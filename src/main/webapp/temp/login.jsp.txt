	<script type="text/javascript">
		$(function() {
			//$("#register-send-btn").attr("disabled", true);
			//替换btn的值
			$("#register-send-btn").click(
					function mail() {
						if (checkuserMail() && checkregisterpassword()
								&& checkregisterRepassword()) {
							$("#register-send-btn").attr("disabled", true);
							$.ajax({
								url : "mailcheck",
								type : "POST",
								datatype : "text",
								data : {
									"mail" : $("#registerMail").val()
								},
								success : function(data) {
									var userInput = document.getElementById("mailcode");
									var strValue = userInput.value;
									if (data == strValue) {
										return true;
									}
								}
							})
							//$("#register-send-btn").attr("disabled", true);
							if (time == 120) {
								this.value = "120秒后重新发送验证码";
								//开启定时器
								id = setInterval(showTime, 1000);
								flag = false;
							}
						}
					});

		});

		var time = 120;
		var flag = true;
		var id;

		function showTime() {
			if (time > 0) {
				time--;
				$("#register-send-btn").val(time + "秒后重新发送验证码");
			} else {
				$("#register-send-btn").val("重新获取邮箱验证码");
				$("#register-send-btn").attr("disabled", false);
				flag = true;
				time = 120;
				clearInterval(id);
			}
		}
	</script>
